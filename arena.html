Usage

        #define ARENA_IMPLEMENTATION
        #include "arena.h"

        arena a = {0};
        unsigned char mem[1024];
        arena_init(&a, mem, 1024);

        {
            char *s = arena_asprintf(&a, "%s, %s!", "Hello", "World");
            puts(s); // Prints "Hello, World!"
        }

        arena_deinit(&a);
